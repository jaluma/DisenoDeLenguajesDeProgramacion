MEDIANTE UNA GRAMÁTICA
LIBRE DE CONTEXTO
start
: instructions
;
instructions
: instruction*
;
instruction
: 'var' def
| 'struct' IDENT '{' defStruct '}' PTO_COMA
| IDENT '(' paramsDef ')' '{' funcionDef funcionSen '}
| IDENT '(' paramsDef ')' ':' tipo '{' funcionDef funcionSen '}'
;
sentence
: 'print' expr PTO_COMA
| 'printsp' expr PTO_COMA
| 'println' PTO_COMA
| 'println' expr PTO_COMA
| expr '=' expr PTO_COMA
| 'return' PTO_COMA
| 'return' expr PTO_COMA
| 'read' expr PTO_COMA
| IDENT '(' params ')' PTO_COMA
| 'if' '(' expr ')' '{' sentences '}'
| 'if' '(' expr ')' '{' sentences '}' 'else' '{' sentences '}'
| 'while' '(' expr ')' '{' sentences '}'
;
expr
: INT_CONSTANT
| REAL_CONSTANT
| CHAR_CONSTANT
| IDENT
| '(' expr ')'
| expr '[' expr ']'
| expr '.' IDENT
| 'cast' '<' tipo '>' '(' expr ')'
| op='!' expr
| expr op=('*' | '/') expr
| expr op=('+' | '-') expr
| expr op=('>' | '>=' | '<' | '<=') expr
| expr op=('==' | '!=') expr
| expr op='&&' expr
| expr op='||' expr
| IDENT '(' params ')'
;
tipo
: INT
| FLOAT
| CHAR
| IDENT
;
paramsDef
: ( IDENT ':' defVarArray ( ',' IDENT ':' defVarArray )* )?
;
funcionDef
: ( 'var' def )*
;
funcionSen
: ( sentence )*
;
defStruct
: ( IDENT ':' defVarArray PTO_COMA )*
;
def
: IDENT ':' defVarArray PTO_COMA
;
defVarArray
: tipo
| '[' INT_CONSTANT ']' defVarArray
;
sentences
: ( sentence )*
;
params
: ( expr ( ',' expr )* )?
;
